<script>
jQuery(window).on('scroll', function () {
    jQuery('.telephony_effect').each(function () {
        var section = jQuery(this);
        var sectionTop = section.offset().top;
        var sectionHeight = section.outerHeight();
        var sectionBottom = sectionTop + sectionHeight;

        var viewportTop = jQuery(window).scrollTop();
        var viewportBottom = viewportTop + jQuery(window).height();

        // Calculate how much of the section is visible
        var visibleTop = Math.max(sectionTop, viewportTop);
        var visibleBottom = Math.min(sectionBottom, viewportBottom);
        var visibleHeight = visibleBottom - visibleTop;

        // Trigger class if 40% or more of the section is visible
        if (visibleHeight > 0 && visibleHeight >= sectionHeight * 0.4) {
            section.addClass('telephony_effect_active');
        } 
    });
});

		var charCode = (evt.which) ? evt.which : evt.keyCode;

			if (charCode == 8 || charCode == 32 || charCode == 9)
